# 项目目录速查

## 后端模块

```
synapse/
├── synapse-shared/                    # 共享内核
│   └── src/main/java/com/synapse/shared/
│       └── exception/                 # DomainException
│
├── synapse-auth-domain/               # 认证领域层
│   └── .../auth/model/                # UserAccount, RoleDefinition, UserId
│   └── .../auth/repository/           # 仓储接口
│
├── synapse-auth-application/          # 认证应用层
│   └── .../auth/port/
│       ├── in/                        # AuthenticationUseCase, UserAdminUseCase
│       ├── out/                       # LoginSessionPort, PasswordHasherPort
│       └── service/                   # AuthApplicationService
│
├── synapse-auth-adapter/              # 认证适配器层
│   └── .../auth/adapter/in/web/       # AuthController, UserAdminController
│   └── .../auth/adapter/out/          # SaTokenLoginSessionAdapter, BCryptPasswordHasherAdapter
│
├── synapse-auth-config/               # 认证配置
│   └── AuthBeanConfig, AuthDataInitializer, SaTokenSecurityConfig
│
├── synapse-kb-domain/                 # 知识库领域层
│   └── .../kb/model/                  # KnowledgeBase, Document, ChatSession, Query, RagContext
│   │                                  # DocumentMetadata, DocumentLifecycleStatus, DocumentSourceType
│   │                                  # ChunkReference, DocumentChunk
│   └── .../kb/repository/             # 仓储接口
│   └── .../kb/service/                # RecursiveChunkingStrategy
│
├── synapse-kb-application/            # 知识库应用层
│   └── .../kb/port/
│       ├── in/                        # IngestDocumentUseCase, QueryKnowledgeBaseUseCase
│       │                              # UpdateDocumentMetadataUseCase (v1 不支持)
│       │                              # SupersedeDocumentUseCase (v1 不支持)
│       ├── out/                       # VectorStorePort, ChunkSearchIndexPort, EmbeddingPort
│       │                              # RetrievalFilter
│       └── service/                   # KnowledgeBaseApplicationService
│
├── synapse-kb-adapter/                # 知识库适配器层
│   └── .../kb/adapter/in/web/         # DocumentController, StreamingQueryController
│   └── .../kb/adapter/out/            # MilvusVectorStoreAdapter (v3 collection)
│   │                                  # MongoChunkSearchIndexAdapter
│   │                                  # OllamaEmbeddingAdapter, OllamaQueryRewriteAdapter
│   │                                  # OllamaStreamingLlmAdapter
│   │                                  # ApacheTikaDocumentParserAdapter
│   │                                  # GridFsDocumentContentStoreAdapter
│   └── .../kb/adapter/out/persistence/entity/
│                                      # DocumentDocument, ChatMessageDocument
│                                      # ChatReferenceDocument, ChunkIndexDocument
│                                      # ChunkPostingDocument, ChunkCorpusStatsDocument
│
├── synapse-kb-config/                 # 知识库配置
│   └── KnowledgeBaseBeanConfig
│
└── synapse-kb-bootstrap/              # 启动入口
    └── SynapseApplication.java
    └── resources/application.yaml     # 所有配置
```

## 前端模块

```
synapse-frontend/
├── src/
│   ├── api/                           # API 客户端
│   │   ├── client.ts                  # axios 配置
│   │   └── query.ts                   # SSE 手动解析
│   ├── stores/                        # Pinia Store
│   │   └── chat.ts                    # 竞态防护、Token 缓冲
│   ├── composables/                   # 可复用逻辑
│   │   └── usePolling.ts              # 轮询 Hook
│   ├── views/                         # 页面
│   ├── components/                    # 通用组件
│   └── router/                        # 路由配置
```
